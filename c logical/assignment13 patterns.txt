1)print A with star pattern:-

#include<stdio.h>
int main(){
    for(int i=1;i<=4;i++){
        for(int j=1;j<=4;j++){
            if(i==1 ||  j==4 || j==1 || i==3){
               printf("*"); 
            }
            else{
                printf(" ");
            }
        }
        printf("\n");
    }
    return 0;
}
